## 组件

- 解耦，每个组件为单一的项目，对外只提供接口
- 复用、扩展性强

```javascript
//组件构建
Vue.component('list-item',{
	props:['data'],
  template:'<li>{{datas.name}}</li>'
})

//模板
<div id="app">
  <ol>
      <list-item 
        v-for="(item,index) in groceryList"
        :data="item"
        :key="index"> 
      <list-item>
  </ol>
</div>

//实例化
var app = new Vue({
  el: '#app',
  data: {
    groceryList: [
      { id: 0, name: '蔬菜' },
      { id: 1, name: '奶酪' },
      { id: 2, name: '随便其它什么人吃的东西' }
    ]
  }
})
```
